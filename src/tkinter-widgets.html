Hoverable Button and events

Creating a button which changes colors on hover

from tkinter import *
root = Tk()
def ChangeBG(event,color):
event.widget.config(background=color)
# doing event.widget was not necessary
button = Button(root, text="Hoverable Button", bg="red",fg="white")
button.pack()
button.bind("<Enter>",func =lambda e: ChangeBG(e,"grey"))
button.bind("<Leave>",func =lambda e: ChangeBG(e,"red"))
root.mainloop()

A Class For Making An Hoverable Button

class mCnButton(Button):
def __init__(self,master,**kwargs):
super(mCnButton,self).__init__(master
,**kwargs)
self.normal = {"background":kwargs["bg"],
"foreground":kwargs["fg"]}
self.onHover = {"background":"black",
"foreground":"white"}
def MakeHoverable(self):
self.bind("<Enter>",func = lambda e:
self.config(**self.onHover))
self.bind("<Leave>",func = lambda e:
self.config(**self.normal))
